  --name ehc.api.v1 \
  --network=eq_historian_network \

  -v /usr/local/EquipmentHistorian/logs:/equipmentHistorianLogs \
  -v /usr/local/EquipmentHistorian/conf:/equipmentHistorianConf \

   api.historian.deadletter:latest
